        function setQuickFilter(type) {
            // Remove active class from all tabs
            document.querySelectorAll('.tab-button').forEach(btn => {
                btn.classList.remove('active');
            });
            
            // Add active class to the correct tab
            const clickedButton = event ? event.target : document.querySelector(`[onclick*="${type}"]`);
            if (clickedButton) {
                clickedButton.classList.add('active');
            }
            
            const today = new Date();
            let startDate, endDate;
            
            switch(type) {
                case 'today':
                    startDate = endDate = today;
                    break;
                case 'week':
                    startDate = new Date(today.getTime() - (7 * 24 * 60 * 60 * 1000));
                    endDate = today;
                    break;
                case 'month':
                    startDate = new Date(today.getFullYear(), today.getMonth(), 1);
                    endDate = today;
                    break;
                case 'fc-reversa':
                    startDate = new Date(today.getTime() - (30 * 24 * 60 * 60 * 1000));
                    endDate = today;
                    break;
            }
            
            document.getElementById('startDate').value = startDate.toISOString().split('T')[0];
            document.getElementById('endDate').value = endDate.toISOString().split('T')[0];
            
            filterHistoricalData(type === 'fc-reversa');
        }

        function filterHistoricalData(fcReversaOnly = false) {
            const startDate = new Date(document.getElementById('startDate').value);
            const endDate = new Date(document.getElementById('endDate').value);
            const content = document.getElementById('historicalContent');
            
            if (!startDate || !endDate) {
                showToast('Por favor, selecione as datas de início e fim', 'error');
                return;
            }
            
            if (startDate > endDate) {
                showToast('A data inicial não pode ser maior que a data final', 'error');
                return;
            }
            
            let filteredPackages = packages.filter(p => {
                const packageDate = new Date(p.timestamp);
                return packageDate >= startDate && packageDate <= endDate;
            });
            
            if (fcReversaOnly) {
                filteredPackages = filteredPackages.filter(p => 
                    p.acao === 'FC' || p.acao === 'Reversa'
                );
            }
            
            generateHistoricalReport(content, filteredPackages, startDate, endDate, fcReversaOnly);
        }

            const startDate = new Date(document.getElementById('startDate').value);
            const endDate = new Date(document.getElementById('endDate').value);
            
            const filteredPackages = packages.filter(p => {
                const packageDate = new Date(p.timestamp);
                return packageDate >= startDate && packageDate <= endDate;
            });
            
            const filename = `auditoria-historico-${startDate.toLocaleDateString('pt-BR').replace(/\//g, '-')}-a-${endDate.toLocaleDateString('pt-BR').replace(/\//g, '-')}.csv`;
            
            const csvContent = [
                ['TBR', 'Motivo', 'Ação', 'Data Reprocesso', 'Observações', 'Registrado'],
                ...filteredPackages.map(p => [
                    p.tbr,
                    p.motivo,
                    p.acao,
                    p.reprocessDate || '',
                    p.observacoes || '',
                    new Date(p.timestamp).toLocaleString('pt-BR')
                ])
            ].map(row => row.join(';')).join('\n');
            
            downloadCSV(csvContent, filename);
            showToast('Relatório histórico exportado com sucesso!', 'success');
        }

        function downloadCSV(csvContent, filename) {
            const blob = new Blob(['\ufeff' + csvContent], { type: 'text/csv;charset=utf-8' });
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            window.URL.revokeObjectURL(url);
        }

        function showToast(message, type = 'success') {
            const toast = document.getElementById('successToast');
            const span = toast.querySelector('span');
            span.textContent = message;
            
            const colorClasses = {
                success: 'from-green-400 to-blue-500',
                error: 'from-red-400 to-pink-500',
                info: 'from-blue-400 to-purple-500'
            };
            
            toast.className = `fixed top-6 right-6 bg-gradient-to-r ${colorClasses[type]} text-white px-8 py-4 rounded-2xl shadow-2xl z-50 transform transition-all duration-500`;
            
            toast.classList.remove('hidden');
            toast.style.transform = 'translateX(100%)';
            
            setTimeout(() => {
                toast.style.transform = 'translateX(0)';
            }, 100);
            
            setTimeout(() => {
                toast.style.transform = 'translateX(100%)';
                setTimeout(() => {
                    toast.classList.add('hidden');
                }, 300);
            }, 3000);
        }
    </script>
